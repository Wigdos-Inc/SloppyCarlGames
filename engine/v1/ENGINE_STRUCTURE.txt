engine/
  v1/
    Bootup.js // Initializes engine boot sequence and startup tasks

    core/
      config.js // Global engine configuration values (and settings application)
      ini.js // Initialize exportable ENGINE API for games.
      meta.js // Engine-wide utilities (functions, constants, APIs, etc)
      validate.js // Validation utilities for game payloads.
      normalize.js // Data normalization of validated game payloads for engine use.

    builder/
      NewBoss.js // Builder for creating boss entities
      NewEntity.js // Builder for generic entity model creation (anything that supports being animated)
      NewLevel.js // Builder for creating levels from incoming data (uses most other builders)
      NewObject.js // Builder of 3D meshes (mainly to be used by other builders)
      NewObstacle.js // Builder of obstacles
      NewTexture.js // Procedurally generated texture mapping onto meshes.
      NewUI.js // Builder of UI components
      NewScatter.js // Procedurally generated static object scatter models placed onto meshes.
      templates/
        characters.json // Character template definitions for builder
        enemies.json // Enemy template definitions
        obstacles.json // Obstacle template definitions
        projectiles.json // Projectile template definitions

    handlers/
      Controls.js // Handles inputs from ui/level cache or sends inputs to game if unconsumed (input manager)
      Cutscene.js // Cutscene orchestration handler (cutscene manager)
      Render.js // Displays everything to the game (end of all visual pipelines)
      Sound.js // Sound manager and helpers
      UI.js // UI handler (UI state manager)

      game/
        Animation.js // Level entity animation handler based on entity states
        Boss.js // Boss behavior handler
        Camera.js // Camera management for levels and cutscenes
        Collectible.js // Collectible item behavior
        Enemy.js // Enemy behavior handler
        Level.js // Level lifecycle and orchestration (entire level state manager)
        Physics.js // Game physics glue for handlers (physics pipeline manager)
        Texture.js // Custom texture and sprite loading onto meshes.

      menu/
        Splash.js // Manages splash screens on game startup
        LoadScreen.js // Loading screen logic
        Credits.js // Credits screen UI/behavior

    cutscene/
      Actors.js // Cutscene actor definitions, manager and helpers
      Animation.js // Cutscene-specific animation handler
      AudioSync.js // Sync audio with cutscene timelines (make sure audio plays at right time and speed)
      Scene.js // Cutscene scene management
      Master.js // Cutscene state manager.

    math/
      Curves.js // Curve utilities (e.g., bezier helpers)
      Forces.js // Force calculation helpers
      Vector3.js // 3D vector math utilities
      Utilities.js // Utility math functions (includes Unit classes).

    physics/
      Buoyancy.js // Buoyancy force calculations
      Collision.js // Collision detection utilities
      Correction.js // Collision correction/response helpers (e.g. adjusting player angle on slopes)
      Current.js // Movement adjustments while in underwater currents
      Gravity.js // Gravity helpers
      Resistance.js // Air/Water Resistance/drag computations

    player/
      Abilities.js // Player ability definitions and applications.
      Animation.js // Player-specific animation handler.
      Master.js // Player master controller (player state manager)
      Model.js // Player model/data representation (uses NewEntity builder for player model)
      Movement.js // Movement handler for player
      characters.json // Player character presets

Notes:
- Builders create 2D HTML elements and 3D WebGL2 meshes and are called by state managers.
- Helpers are state managers and helper functions to simplify pipeline.
  - Games are only allowed to interact with helpers accepting payloads (e.g. UI.js or Level.js).
  - Game payloads must first be validated and normalized through validate.js before the data is usable.
- Render.js is the only file allowed to display visual elements to the game.
